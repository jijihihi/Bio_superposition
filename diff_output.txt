+def mlp_gene_prediction(X, Y, labels, args, seed, rng=None, tag=""):
+    """
+    5-fold Stratified CV MLP for gene prediction with anti-overfitting.
+    X: image PCA features (n, gene_pred_pca_dim)
+    Y: soft target gene expression (n, n_genes)
+    labels: genotype labels for stratified split
+    args: namespace with MLP hyperparams (gene_mlp_* or mlp_* prefixed)
+    Returns dict with per-gene R짼 (averaged across folds), predictions, etc.
+    """
+    import torch
+    import torch.nn as nn
+    from torch.utils.data import TensorDataset, DataLoader
+    from sklearn.model_selection import StratifiedKFold, train_test_split
+
+    n_samples, n_input = X.shape
+    n_genes = Y.shape[1]
+    hidden = getattr(args, "gene_mlp_hidden", getattr(args, "mlp_hidden", 128))
+    dropout = getattr(args, "gene_mlp_dropout", getattr(args, "mlp_dropout", 0.4))
+    max_epochs = getattr(args, "gene_mlp_epochs", getattr(args, "mlp_epochs", 200))
+    patience = getattr(args, "gene_mlp_patience", getattr(args, "mlp_patience", 30))
+    lr = getattr(args, "gene_mlp_lr", getattr(args, "mlp_lr", 5e-4))
+    bs = getattr(args, "gene_mlp_batch_size", getattr(args, "mlp_batch_size", 256))
+    ctrl = getattr(args, "img_genotype_ctrl", "Control")
+    n_folds = getattr(args, "gene_mlp_n_folds", 5)
+
+    device = torch.device("cuda" if torch.cuda.is_available() else "cpu")
+
+    def _safe_r2(y_true, y_pred):
+        ss_tot = np.sum((y_true - y_true.mean()) ** 2)
+        if ss_tot < 1e-12:
+            return 0.0
+        ss_res = np.sum((y_true - y_pred) ** 2)
+        return max(1.0 - ss_res / ss_tot, -1.0)
+
+    def _safe_spearman(a, b):
+        if np.std(a) < 1e-12 or np.std(b) < 1e-12:
+            return 0.0
+        return float(spearmanr(a, b)[0])
+

+

+    all_fold_test_rho = []
+    all_fold_train_r2_avg = []
+    all_fold_stopped = []
+    all_fold_val_loss = []
+    n_params_printed = False
+
+    print(f"        MLP {tag}: {n_folds}-fold CV "

+        X_trainval, X_test = X[trainval_idx], X[test_idx]
+        Y_trainval, Y_test = Y[trainval_idx], Y[test_idx]
+        lbl_tv = label_bin[trainval_idx]
+
+        # Internal val split (12.5% of trainval ~ 10% total) for early stopping
+        X_train, X_val, Y_train, Y_val = train_test_split(
+            X_trainval, Y_trainval,
+            test_size=0.125, random_state=seed + fold_i, stratify=lbl_tv)
+
+        # Standardize (fit on train only)
+        mu_x = X_train.mean(axis=0, keepdims=True)
+        sd_x = X_train.std(axis=0, keepdims=True) + 1e-8
+        mu_y = Y_train.mean(axis=0, keepdims=True)
+        sd_y = Y_train.std(axis=0, keepdims=True) + 1e-8
+
+        def _to_t(arr):
+            return torch.tensor(arr, dtype=torch.float32)
+
+        Xt = _to_t((X_train - mu_x) / sd_x)
+        Xv = _to_t((X_val - mu_x) / sd_x)
+        Xte = _to_t((X_test - mu_x) / sd_x)
+        Yt = _to_t((Y_train - mu_y) / sd_y)
+        Yv_d = _to_t((Y_val - mu_y) / sd_y).to(device)
+
+        # Model (same architecture as before)
+        model = nn.Sequential(
+            nn.BatchNorm1d(n_input),
+            nn.Linear(n_input, hidden),
+            nn.ReLU(),
+            nn.Dropout(dropout),
+            nn.BatchNorm1d(hidden),
+            nn.Linear(hidden, hidden),
+            nn.ReLU(),
+            nn.Dropout(dropout),
+            nn.BatchNorm1d(hidden),
+            nn.Linear(hidden, n_genes),
+        ).to(device)
+
+        if not n_params_printed:
+            n_params = sum(p.numel() for p in model.parameters() if p.requires_grad)

+            n_params_printed = True
+
+        optimizer = torch.optim.Adam(model.parameters(), lr=lr, weight_decay=1.2e-3)

+
+        ds = TensorDataset(Xt.to(device), Yt.to(device))

+
+        best_val_loss = float('inf')
+        best_state = None

+
+        for ep in range(max_epochs):
+            model.train()
+            for xb, yb in dl:
+                optimizer.zero_grad()
+                loss = criterion(model(xb), yb)

+                torch.nn.utils.clip_grad_norm_(model.parameters(), max_norm=1.0)

+
+            model.eval()
+            with torch.no_grad():

+

+                wait = 0
+            else:

+            if wait >= patience:

+

+
+        if best_state is not None:

+                                  for d in range(n_genes)])

+                                   for d in range(n_genes)])
+
+        all_fold_test_r2.append(fold_test_r2)
+        all_fold_test_rho.append(fold_test_rho)
+        all_fold_train_r2_avg.append(float(np.mean(fold_train_r2)))
+        all_fold_stopped.append(stopped_ep)
+        all_fold_val_loss.append(best_val_loss)
+
+        print(f"          Fold {fold_i+1}/{n_folds}: "
+              f"test R짼={np.mean(fold_test_r2):.4f}, "
+              f"train R짼={np.mean(fold_train_r2):.4f}, "
+              f"stopped@{stopped_ep} "
+              f"(train={X_train.shape[0]}, val={X_val.shape[0]}, "
+              f"test={X_test.shape[0]})")
+
+    # ---- Average across folds ----
+    test_r2 = np.mean(all_fold_test_r2, axis=0)
+    test_rho = np.mean(all_fold_test_rho, axis=0)
+    train_r2_avg = float(np.mean(all_fold_train_r2_avg))
+
+    result = {
+        "test_r2_per_gene": test_r2,

+        "test_rho_per_gene": test_rho,

+    def _fast_ridge_perm_zscore(X, Y, n_folds, seed, ridge_alpha, n_perm):
+        """
+        Massively optimized per-gene z-scored R짼 via precomputed projection.
+
+        Ridge closed-form: 棺 = (X^TX + 慣I)^{-1} X^T Y
+        Key insight: P_fold = X_test @ (X_train^T X_train + 慣I)^{-1} X_train^T
+        only depends on X (not Y). Precompute once per fold, then apply to
+        all permuted Y matrices via fast matrix multiplication.
+
+        800 Ridge fits ??5 matrix inversions + 4000 matmuls = ~100x speedup.
+        """
+        n_samples, n_features = X.shape
+        n_genes = Y.shape[1]
+        kf = KFold(n_splits=n_folds, shuffle=True, random_state=seed)
+
+        # ?? 1. Precompute projection matrices per fold ??
+        fold_info = []  # (train_idx, test_idx, P_fold)
+        for train_idx, test_idx in kf.split(X):
+            X_tr = X[train_idx]   # (n_train, d)
+            X_te = X[test_idx]    # (n_test, d)
+            # Ridge hat matrix: P = X_te @ (X_tr^T X_tr + 慣I)^{-1} X_tr^T
+            XtX = X_tr.T @ X_tr   # (d, d)
+            XtX[np.diag_indices_from(XtX)] += ridge_alpha
+            XtX_inv = np.linalg.inv(XtX)  # (d, d) ??only d횞d, very fast
+            P = X_te @ XtX_inv @ X_tr.T   # (n_test, n_train)
+            fold_info.append((train_idx, test_idx, P))
+
+        # ?? 2. Real prediction ??per-gene R짼 ??
+        all_pred_real = np.zeros_like(Y)
+        for train_idx, test_idx, P in fold_info:
+            all_pred_real[test_idx] = P @ Y[train_idx]  # fast matmul
+        r2_real = np.array([r2_score(Y[:, d], all_pred_real[:, d])
+                            for d in range(n_genes)])
+
+        # ?? 3. Batch permutation ??all at once ??
+        # For each perm: shuffle Y rows, predict via precomputed P, compute R짼
+        rng_perm = np.random.RandomState(seed + 999)
+        null_per_gene_r2 = np.zeros((n_perm, n_genes), dtype=np.float32)
+        null_avg_r2 = np.zeros(n_perm, dtype=np.float64)
+        null_fold_r2s = []
+
+        for pi in range(n_perm):
+            perm_idx = rng_perm.permutation(n_samples)
+            Y_perm = Y[perm_idx]
+            all_pred_perm = np.zeros_like(Y)
+            fold_r2s = []
+            for train_idx, test_idx, P in fold_info:
+                all_pred_perm[test_idx] = P @ Y_perm[train_idx]
+                # Vectorized per-fold avg R짼
+                Yt = Y_perm[test_idx]  # (n_test, n_genes)
+                Yp = all_pred_perm[test_idx]
+                ss_res = np.sum((Yt - Yp) ** 2, axis=0)   # (n_genes,)
+                ss_tot = np.sum((Yt - Yt.mean(axis=0)) ** 2, axis=0)  # (n_genes,)
+                fold_r2_vec = 1.0 - ss_res / np.maximum(ss_tot, 1e-12)
+                fold_r2s.append(float(fold_r2_vec.mean()))
+
+            # Vectorized global R짼 across all folds
+            ss_res_g = np.sum((Y_perm - all_pred_perm) ** 2, axis=0)
+            ss_tot_g = np.sum((Y_perm - Y_perm.mean(axis=0)) ** 2, axis=0)
+            perm_r2 = 1.0 - ss_res_g / np.maximum(ss_tot_g, 1e-12)
+            null_per_gene_r2[pi] = perm_r2
+            null_avg_r2[pi] = perm_r2.mean()
+            null_fold_r2s.extend(fold_r2s)
+
+            if (pi + 1) % 100 == 0:
+                print(f"        perm {pi+1}/{n_perm}...")
+
+        null_fold_r2s = np.array(null_fold_r2s)
+        return r2_real, null_per_gene_r2, null_avg_r2, null_fold_r2s
